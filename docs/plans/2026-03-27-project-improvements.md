def generate_config(self, domains: Optional[List[str]] = None) -> str:
    if domains is None:
        domains = self.load_domains()
    
    if not domains:
        return ""
    
    lines = [
        "# AI Domains DNS Spoofing (IPv4 + IPv6)",
        "# Generated by dns_spoofing.py",
        f"# Domains count: {len(domains)}",
        "",
    ]

    config = '\n'.join(lines)
    return config

def save_ipset(setname: str, save_path: str) -> Tuple[bool, str]:
    """Save ipset to file for later restore"""
    try:
        with open(save_path, 'w') as f:
            result = subprocess.run(
                ['ipset', 'list', setname],
